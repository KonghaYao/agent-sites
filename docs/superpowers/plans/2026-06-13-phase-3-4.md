// ── 数据库资源 ──

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Database {
    pub id: String,
    pub name: String,
    pub turso_db_name: String,
    pub turso_url: String,
    pub turso_token: String,
    pub region: String,
    pub status: String,
    pub created_at: String,
    pub updated_at: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SiteDatabaseBinding {
    pub site_id: String,
    pub database_id: String,
    pub created_at: String,
}

/// 创建数据库记录
pub async fn create_database(
    pool: &SqlitePool,
    name: &str,
    turso_db_name: &str,
    turso_url: &str,
    turso_token: &str,
    region: &str,
) -> Result<Database, sqlx::Error> {
    let now = Utc::now().to_rfc3339();
    let id = Uuid::now_v7().to_string();

    sqlx::query(
        "INSERT INTO databases (id, name, turso_db_name, turso_url, turso_token, region, status, created_at, updated_at) \
         VALUES (?, ?, ?, ?, ?, ?, 'active', ?, ?)",
    )
    .bind(&id)
    .bind(name)
    .bind(turso_db_name)
    .bind(turso_url)
    .bind(turso_token)
    .bind(region)
    .bind(&now)
    .bind(&now)
    .execute(pool)
    .await?;

    let db = get_database(pool, &id).await?.ok_or(sqlx::Error::RowNotFound)?;
    Ok(db)
}

/// 按 ID 查询数据库
pub async fn get_database(pool: &SqlitePool, id: &str) -> Result<Option<Database>, sqlx::Error> {
    let row = sqlx::query_as::<_, DatabaseRow>(
        "SELECT id, name, turso_db_name, turso_url, turso_token, region, status, created_at, updated_at \
         FROM databases WHERE id = ?",
    )
    .bind(id)
    .fetch_optional(pool)
    .await?;

    Ok(row.map(|r| r.into()))
}

/// 列出所有 active 数据库
pub async fn list_databases(pool: &SqlitePool) -> Result<Vec<Database>, sqlx::Error> {
    let rows = sqlx::query_as::<_, DatabaseRow>(
        "SELECT id, name, turso_db_name, turso_url, turso_token, region, status, created_at, updated_at \
         FROM databases WHERE status = 'active' ORDER BY created_at DESC",
    )
    .fetch_all(pool)
    .await?;

    Ok(rows.into_iter().map(|r| r.into()).collect())
}

/// 软删除数据库（标记 inactive）
pub async fn delete_database(pool: &SqlitePool, id: &str) -> Result<bool, sqlx::Error> {
    let now = Utc::now().to_rfc3339();
    let result = sqlx::query("UPDATE databases SET status = 'inactive', updated_at = ? WHERE id = ?")
        .bind(&now)
        .bind(id)
        .execute(pool)
        .await?;

    Ok(result.rows_affected() > 0)
}

/// 按名称查找数据库（用于重名检测）
pub async fn get_database_by_name(pool: &SqlitePool, name: &str) -> Result<Option<Database>, sqlx::Error> {
    let row = sqlx::query_as::<_, DatabaseRow>(
        "SELECT id, name, turso_db_name, turso_url, turso_token, region, status, created_at, updated_at \
         FROM databases WHERE name = ? AND status = 'active'",
    )
    .bind(name)
    .fetch_optional(pool)
    .await?;

    Ok(row.map(|r| r.into()))
}

// ── 站点↔数据库绑定 ──

/// 绑定数据库到站点
pub async fn bind_database(
    pool: &SqlitePool,
    site_id: &str,
    database_id: &str,
) -> Result<bool, sqlx::Error> {
    let now = Utc::now().to_rfc3339();
    let result = sqlx::query(
        "INSERT INTO site_database_bindings (site_id, database_id, created_at) VALUES (?, ?, ?)",
    )
    .bind(site_id)
    .bind(database_id)
    .bind(&now)
    .execute(pool)
    .await?;

    Ok(result.rows_affected() > 0)
}

/// 列出站点绑定的所有数据库
pub async fn list_bindings_for_site(
    pool: &SqlitePool,
    site_id: &str,
) -> Result<Vec<SiteDatabaseBinding>, sqlx::Error> {
    let rows = sqlx::query_as::<_, BindingRow>(
        "SELECT site_id, database_id, created_at FROM site_database_bindings WHERE site_id = ? ORDER BY created_at ASC",
    )
    .bind(site_id)
    .fetch_all(pool)
    .await?;

    Ok(rows.into_iter().map(|r| r.into()).collect())
}

/// 解绑
pub async fn unbind_database(
    pool: &SqlitePool,
    site_id: &str,
    database_id: &str,
) -> Result<bool, sqlx::Error> {
    let result = sqlx::query(
        "DELETE FROM site_database_bindings WHERE site_id = ? AND database_id = ?",
    )
    .bind(site_id)
    .bind(database_id)
    .execute(pool)
    .await?;

    Ok(result.rows_affected() > 0)
}

/// 查询站点绑定的所有数据库详情（用于环境变量注入）
pub async fn list_database_details_for_site(
    pool: &SqlitePool,
    site_id: &str,
) -> Result<Vec<Database>, sqlx::Error> {
    let rows = sqlx::query_as::<_, DatabaseRow>(
        "SELECT d.id, d.name, d.turso_db_name, d.turso_url, d.turso_token, d.region, d.status, d.created_at, d.updated_at \
         FROM databases d \
         INNER JOIN site_database_bindings b ON d.id = b.database_id \
         WHERE b.site_id = ? AND d.status = 'active' \
         ORDER BY d.name ASC",
    )
    .bind(site_id)
    .fetch_all(pool)
    .await?;

    Ok(rows.into_iter().map(|r| r.into()).collect())
}

struct DatabaseRow {
    id: String,

            status: r.status,

#[derive(Debug, Deserialize)]
struct TursoCreateDbResponse {
    database: TursoDatabaseInfo,
}

#[derive(Debug, Deserialize)]
struct TursoDatabaseInfo {
    #[serde(rename = "Name")]
    name: String,
    #[serde(rename = "Hostname")]
    hostname: String,
}

/// Turso API 创建 token 响应
#[derive(Debug, Deserialize)]
struct TursoCreateTokenResponse {
    jwt: String,
}

impl TursoClient {
    pub fn new(api_url: String, api_token: String, org: String) -> Self {
        Self {
            api_url,
            api_token,
            org,
            client: reqwest::Client::new(),
        }
    }

    /// 创建 Turso 数据库，返回 (turso_db_name, turso_url, token)
    pub async fn create_database(
        &self,
        name: &str,
        region: &str,
    ) -> Result<(String, String, String), AppError> {
        // Step 1: 创建数据库
        let create_url = format!("{}/v1/organizations/{}/databases", self.api_url, self.org);
        let body = serde_json::json!({
            "name": name,
            "group": region,
        });

        let resp = self
            .client
            .post(&create_url)
            .header("Authorization", format!("Bearer {}", self.api_token))
            .json(&body)
            .send()
            .await
            .map_err(|e| AppError::Internal(format!("Turso 创建数据库请求失败: {e}")))?;

        if !resp.status().is_success() {
            let status = resp.status();
            let text = resp.text().await.unwrap_or_default();
            tracing::error!(status = %status, body = %text, "Turso 创建数据库失败");
            return Err(AppError::Internal("数据库服务暂不可用".to_string()));
        }

        let create_resp: TursoCreateDbResponse = resp
            .json()
            .await
            .map_err(|e| AppError::Internal(format!("Turso 响应解析失败: {e}")))?;

        let turso_db_name = create_resp.database.name;
        let turso_url = self.format_turso_url(&create_resp.database.hostname);

        // Step 2: 创建 auth token
        let token_url = format!(
            "{}/v1/organizations/{}/databases/{}/auth-tokens",
            self.api_url, self.org, turso_db_name
        );

        let token_resp = self
            .client
            .post(&token_url)
            .header("Authorization", format!("Bearer {}", self.api_token))
            .send()
            .await
            .map_err(|e| AppError::Internal(format!("Turso 创建 token 请求失败: {e}")))?;

        if !token_resp.status().is_success() {
            let status = token_resp.status();
            let text = token_resp.text().await.unwrap_or_default();
            tracing::error!(status = %status, body = %text, "Turso 创建 token 失败");
            return Err(AppError::Internal("数据库服务暂不可用".to_string()));
        }

        let token_data: TursoCreateTokenResponse = token_resp
            .json()
            .await
            .map_err(|e| AppError::Internal(format!("Turso token 响应解析失败: {e}")))?;

        Ok((turso_db_name, turso_url, token_data.jwt))
    }

    /// 删除 Turso 数据库（best-effort，失败只打日志不报错）
    pub async fn delete_database(&self, turso_db_name: &str) {
        let url = format!(
            "{}/v1/organizations/{}/databases/{}",
            self.api_url, self.org, turso_db_name
        );

        match self
            .client
            .delete(&url)
            .header("Authorization", format!("Bearer {}", self.api_token))
            .send()
            .await
        {
            Ok(resp) if resp.status().is_success() => {
                tracing::info!(turso_db_name = %turso_db_name, "Turso 数据库已删除");
            }
            Ok(resp) => {
                let status = resp.status();
                let text = resp.text().await.unwrap_or_default();
                tracing::warn!(status = %status, body = %text, turso_db_name = %turso_db_name, "Turso 删除数据库失败");
            }
            Err(e) => {
                tracing::warn!(error = %e, turso_db_name = %turso_db_name, "Turso 删除数据库请求失败");
            }
        }
    }

    /// 构造 libsql:// 连接 URL
    pub fn format_turso_url(&self, hostname: &str) -> String {
        format!("libsql://{hostname}")
    }
}

/// POST /api/databases — 创建 Turso 数据库
pub async fn create_database(

    Json(input): Json<CreateDatabaseInput>,

    if db::models::get_database_by_name(&state.db, &input.name)
        .await?
        .is_some()
    {

        .await?;

        &state.db,
        &input.name,
        &turso_db_name,
        &turso_url,
        &turso_token,
        &input.region,
    )
    .await?;

    Ok((StatusCode::CREATED, Json(ApiResponse::ok(database))))

pub async fn list_databases(
    State(state): State<Arc<AppState>>,

    let databases = db::models::list_databases(&state.db).await?;
    Ok(Json(ApiResponse::ok(databases)))
}

pub async fn get_database(
    State(state): State<Arc<AppState>>,
    Path(id): Path<String>,
) -> Result<impl IntoResponse, AppError> {
    let database = db::models::get_database(&state.db, &id)
        .await?
        .ok_or(AppError::NotFound)?;
    Ok(Json(ApiResponse::ok(database)))

/// DELETE /api/databases/:id — 删除数据库（软删除 + 调 Turso 删除）
pub async fn delete_database(
    State(state): State<Arc<AppState>>,
    Path(id): Path<String>,
) -> Result<impl IntoResponse, AppError> {
    let db_record = db::models::get_database(&state.db, &id)
        .await?
        .ok_or(AppError::NotFound)?;

    // 软删除（标记 inactive）
    db::models::delete_database(&state.db, &id).await?;

    // 清理绑定关系
    // （通过 CASCADE 或手动清理 — SQLite 默认不级联删除，但我们的 FK 是声明式的，需显式清理）
    // 此处选择容忍：如果后续需要清理绑定，可加一步；当前保留绑定记录用于审计

    // 异步删 Turso 资源（best-effort）
    let turso_db_name = db_record.turso_db_name.clone();
    let turso_client = state.turso_client.clone_for_task();
    tokio::spawn(async move {
        turso_client.delete_database(&turso_db_name).await;
    });

    Ok(StatusCode::NO_CONTENT)
}

    Json(input): Json<BindDatabaseInput>,

    if db_record.status != "active" {
        return Err(AppError::BadRequest("数据库不可用".to_string()));
    }

    let bound = db::models::bind_database(&state.db, &site_id, &input.database_id).await;

            Json(ApiResponse::ok(serde_json::json!({"bound": true}))),

                Err(AppError::Conflict("该数据库已绑定此站点".to_string()))
            } else {
                Err(AppError::Database(e))
            }
        }

```rust
use crate::db;
use crate::error::{ApiResponse, AppError};
use crate::AppState;
use axum::extract::{Multipart, Path, State};
use axum::http::StatusCode;
use axum::response::IntoResponse;
use axum::Json;
use flate2::read::GzDecoder;
use std::io::Read;
use std::sync::Arc;

/// POST /api/sites/:id/deploy — 上传 tar.gz，创建版本并激活
pub async fn deploy_site(
    State(state): State<Arc<AppState>>,
    Path(site_id): Path<String>,
    mut multipart: Multipart,
) -> Result<impl IntoResponse, AppError> {
    // 验证站点存在
    let site = db::models::get_site(&state.db, &site_id)
        .await?
        .ok_or(AppError::NotFound)?;

    // 提取文件字段
    let mut tar_gz_data: Option<Vec<u8>> = None;
    let mut filename: Option<String> = None;

    while let Some(field) = multipart
        .next_field()
        .await
        .map_err(|e| AppError::BadRequest(format!("multipart 解析失败: {e}")))?
    {
        let name = field.name().unwrap_or("").to_string();
        if name == "file" {
            filename = field.file_name().map(|s| s.to_string());
            let data = field
                .bytes()
                .await
                .map_err(|e| AppError::BadRequest(format!("读取文件失败: {e}")))?;
            tar_gz_data = Some(data.to_vec());
            break;
        }
    }

    let filename = filename.ok_or_else(|| AppError::BadRequest("缺少 file 字段".to_string()))?;
    let tar_gz_data =
        tar_gz_data.ok_or_else(|| AppError::BadRequest("文件内容为空".to_string()))?;

    // 校验扩展名
    if !filename.ends_with(".tar.gz") {
        return Err(AppError::BadRequest("格式错误：仅支持 .tar.gz".to_string()));
    }

    // 校验大小
    let max_bytes = state.max_upload_size_mb * 1024 * 1024;
    if tar_gz_data.len() as u64 > max_bytes {
        return Err(AppError::PayloadTooLarge(state.max_upload_size_mb));
    }

    // 解压 tar.gz → 检查 main.ts
    let gz = GzDecoder::new(&tar_gz_data[..]);
    let mut archive = tar::Archive::new(gz);
    let mut has_main_ts = false;

    // 第一遍扫描：验证 main.ts 存在
    let entries = archive
        .entries()
        .map_err(|e| AppError::BadRequest(format!("tar 解析失败: {e}")))?;
    for entry in entries {
        let entry = entry.map_err(|e| AppError::BadRequest(format!("tar 条目错误: {e}")))?;
        let path = entry.path().map_err(|e| AppError::BadRequest(format!("路径解析失败: {e}")))?;
        if path.to_string_lossy() == "main.ts" {
            has_main_ts = true;
            break;
        }
    }

    if !has_main_ts {
        return Err(AppError::BadRequest(
            "入口文件缺失：main.ts 必须位于根目录".to_string(),
        ));
    }

    // 创建版本记录
    let version = db::models::create_version(&state.db, &site_id).await?;

    // 解压到版本目录
    let version_dir = state.storage_dir.join(&site_id).join(&version.code_dir);
    tokio::fs::create_dir_all(&version_dir)
        .await
        .map_err(|e| AppError::Internal(format!("创建版本目录失败: {e}")))?;

    let gz2 = GzDecoder::new(&tar_gz_data[..]);
    let mut archive2 = tar::Archive::new(gz2);
    archive2
        .unpack(&version_dir)
        .map_err(|e| AppError::Internal(format!("解包失败: {e}")))?;

    // 激活版本
    db::models::set_active_version(&state.db, &site_id, &version.id).await?;

    // 停止旧 Deno 进程（如有）
    if state.process_manager.is_running(&site_id) {
        state.process_manager.stop(&state.db, &site_id).await?;
    }

    Ok((
        StatusCode::CREATED,
        Json(ApiResponse::ok(serde_json::json!({
            "version": version,
            "activated": true
        }))),
    ))
}
```

use std::collections::HashMap;
use std::process::Stdio;
use tokio::process::{Child, Command};

/// 跨平台 Deno 进程启动（通过 bash -c / cmd /C）
#[cfg(unix)]
pub fn spawn_deno(
    main_ts_path: &str,
    port: u16,
    deno_path: &str,
    extra_env: &HashMap<String, String>,
) -> std::io::Result<Child> {
    let cmd_str = format!(
        "{} run --allow-net --allow-env --allow-read {}",
        deno_path, main_ts_path
    );
    let mut cmd = Command::new("bash");
    cmd.arg("-c").arg(&cmd_str);
    cmd.env("PORT", port.to_string());
    for (key, value) in extra_env {
        cmd.env(key, value);
    }
    cmd.stdin(Stdio::null());
    cmd.stdout(Stdio::piped());
    cmd.stderr(Stdio::piped());
    cmd.kill_on_drop(true);
    cmd.spawn()
}

#[cfg(windows)]
pub fn spawn_deno(
    main_ts_path: &str,
    port: u16,
    deno_path: &str,
    extra_env: &HashMap<String, String>,
) -> std::io::Result<Child> {

        deno_path, main_ts_path
    );

    cmd.env("PORT", port.to_string());
    for (key, value) in extra_env {
        cmd.env(key, value);
    }
    cmd.stdin(Stdio::null());
    cmd.stdout(Stdio::piped());
    cmd.stderr(Stdio::piped());
    cmd.kill_on_drop(true);
    cmd.spawn()
}

## 站点代码中使用数据库

Deno 站点启动时，平台自动注入环境变量。每个绑定到站点的数据库提供两个环境变量：

```
TURSO_DB_<NAME>_URL=libsql://xxx.turso.io
TURSO_DB_<NAME>_TOKEN=eyJ...
```

`<NAME>` 为数据库 name 字段的大写形式（非字母数字替换为 `_`）。

站点 `main.ts` 中使用：

```typescript
import { createClient } from "@libsql/client";

// name 为 "my-db" 时，环境变量名为 TURSO_DB_MY_DB_URL / TURSO_DB_MY_DB_TOKEN
const db = createClient({
    url: Deno.env.get("TURSO_DB_MY_DB_URL")!,
    authToken: Deno.env.get("TURSO_DB_MY_DB_TOKEN")!,
});

// 处理 HTTP 请求
Deno.serve({ port: parseInt(Deno.env.get("PORT") || "3000") }, async (req) => {
    const url = new URL(req.url);

    if (url.pathname === "/api/health") {
        return new Response("ok");
    }

    if (url.pathname === "/api/users") {
        const rs = await db.execute("SELECT * FROM users");
        return Response.json(rs.rows);
    }

    return new Response("not found", { status: 404 });
});
```

## 常见问题

### 部署返回 400 "入口文件缺失"

tar.gz 包根目录没有 `main.ts`。确保打包命令在项目根目录执行：

```bash
tar czf site.tar.gz main.ts public/
```

### 部署返回 413 "文件过大"

包超过大小限制（默认 50MB）。减少文件或调整平台配置 `MAX_UPLOAD_SIZE_MB`。

### 数据库绑定后站点无法连接

1. 确认站点 Deno 进程已重启（部署新版本会自动重启）
2. 确认数据库中状态为 `active`
3. 确认站点代码中使用的环境变量名与数据库 name 匹配（大写、非字母数字转 `_`）
```
